❌ Naive Recursive
let rec factorial n =
  if n <= 1 then 1
  else n * factorial (n - 1)

(* factorial 5
   = 5 * factorial 4
   = 5 * (4 * factorial 3)
   = 5 * (4 * (3 * factorial 2))
   = 5 * (4 * (3 * (2 * 1)))
   Stack frames: 5 *)
✅ Tail Recursive
let factorial n =
  let rec aux n acc =
    if n <= 1 then acc
    else aux (n - 1) (n * acc)
  in
  aux n 1

(* aux 5 1 → aux 4 5 → aux 3 20
   → aux 2 60 → aux 1 120 → 120
   Stack frames: 1 *)

Transformation Steps

1
Identify the operation after the recursive call: n * factorial (n-1) — multiplication happens after return.
2
Add an accumulator parameter acc to carry the partial result. Initialize with the identity value (1 for multiplication).
3
Move the operation into the recursive call: aux (n-1) (n * acc) — now the call is in tail position.
4
Base case returns acc instead of the identity value.
5
Wrap in a non-recursive outer function to hide the accumulator from the API.

❌ Naive Recursive
let rec fib n =
  if n <= 1 then n
  else fib (n - 1) + fib (n - 2)

(* Exponential tree of calls!
   fib 5 calls fib 4 + fib 3
   fib 4 calls fib 3 + fib 2
   ... massive redundancy *)
✅ Tail Recursive
let fib n =
  let rec aux n a b =
    if n = 0 then a
    else aux (n - 1) b (a + b)
  in
  aux n 0 1

(* aux 5 0 1 → aux 4 1 1 → aux 3 1 2
   → aux 2 2 3 → aux 1 3 5 → aux 0 5 8
   → 5.  Linear, O(1) stack! *)

Transformation Steps

1
Two recursive calls means we need two accumulators: a (current fib) and b (next fib).
2
Each step shifts the window: a → b, b → a + b.
3
Count down from n; when n = 0, a holds the answer.
4
This also fixes the exponential time — each value computed exactly once.

let rec length = function

let length lst =
  let rec aux acc = function
    | [] -> acc
    | _ :: tl -> aux (acc + 1) tl
  in
  aux 0 lst

Transformation Steps

1
1 + length tl — addition after return, not tail-recursive.
2
Add acc starting at 0. Increment before recursing: aux (acc + 1) tl.
3
Base case returns acc.

❌ Naive Recursive
let rec reverse = function
  | [] -> []
  | hd :: tl -> reverse tl @ [hd]

(* @ is O(n), called n times = O(n²) *)
✅ Tail Recursive
let reverse lst =
  let rec aux acc = function
    | [] -> acc
    | hd :: tl -> aux (hd :: acc) tl
  in
  aux [] lst

(* :: is O(1), called n times = O(n) *)

Transformation Steps

1
The naive version appends to end (@ [hd]) — quadratic.
2
Accumulator is a list; prepend each element: hd :: acc. Prepending naturally reverses!
3
This is how List.rev is implemented in OCaml's standard library.

❌ Naive Recursive
let rec map f = function
  | [] -> []
  | hd :: tl -> f hd :: map f tl
✅ Tail Recursive
let map f lst =
  let rec aux acc = function
    | [] -> List.rev acc
    | hd :: tl -> aux (f hd :: acc) tl
  in
  aux [] lst

(* Builds reversed result, then
   reverses at the end. Still O(n). *)

Transformation Steps

1
f hd :: map f tl — cons after recursive call, not tail-recursive.
2
Accumulate results with prepend: f hd :: acc. This reverses the output.
3
Call List.rev acc at the base case to restore correct order.
4
Pattern: When building a list tail-recursively, you almost always need a final List.rev.

(* This is essentially List.fold_left (+) 0 *)

Transformation Steps

1
Same pattern as factorial/length: move + into the recursive call via accumulator.
2
Insight: Tail-recursive sum is just fold_left! Most accumulator patterns generalize to folds.

❌ Naive Recursive
let rec flatten = function
  | [] -> []
  | hd :: tl -> hd @ flatten tl
✅ Tail Recursive
let flatten lst =
  let rec aux acc = function
    | [] -> List.rev acc
    | hd :: tl ->
      let acc' = List.fold_left
        (fun a x -> x :: a) acc hd
      in
      aux acc' tl
  in
  aux [] lst

Transformation Steps

1
Each sub-list's elements are prepended onto the accumulator (reversed).
2
List.fold_left handles inner list traversal — already tail-recursive.
3
Final List.rev restores order.

✅ Already Tail Recursive!
let rec gcd a b =
  if b = 0 then a
  else gcd b (a mod b)

(* The recursive call IS the last thing —
   no operation wraps it.
   gcd 48 18 → gcd 18 12 → gcd 12 6
   → gcd 6 0 → 6 *)
🤔 How to Recognize Tail Position
(* TAIL position — call IS the result: *)
else gcd b (a mod b)   (* ✅ *)

(* NOT tail position — work after call: *)
else 1 + f (n-1)       (* ❌ add *)
else f (n-1) @ [x]     (* ❌ append *)
else x :: f tl        (* ❌ cons *)

(* Key test: if you removed the call
   and put a value there, would the
   function still return correctly? *)

Why GCD is Already Tail-Recursive

1
The recursive call gcd b (a mod b) is the entire return value — nothing wraps it.
2
OCaml's compiler recognizes this and reuses the current stack frame (like a loop).
3
Lesson: Not all recursion needs conversion! Check if the call is already in tail position.
